# Vosi v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use tre:subagent-driven-development (recommended) or tre:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a dual-platform (Windows/macOS) offline desktop voice input app that transcribes hold-to-talk speech via FunASR ONNX models running on sherpa-onnx, then injects text into the focused input field.

**Architecture:** Tauri 2 hosts a Rust core that owns audio capture, VAD, offline Paraformer ASR, punctuation, post-processing (hotword + ITN), and platform text injection. Models ship inside the installer and are verified on first launch. No Python at runtime; no outbound network.

**Tech Stack:** Tauri 2, Rust, `sherpa-onnx` crate 1.13.x, `cpal`, `rdev`, `serde`/`toml`, vanilla TS settings UI

**Spec reference:** `docs/specs/2026-06-05-vosi-v01-design.md`

```rust
// src-tauri/src/post/hotword.rs
use std::collections::HashSet;

pub struct HotwordReplacer {
    words: Vec<String>,
}

impl HotwordReplacer {
    pub fn from_lines(lines: impl IntoIterator<Item = String>) -> Self {
        let mut words: Vec<String> = lines
            .into_iter()
            .map(|l| l.trim().to_string())
            .filter(|l| !l.is_empty())
            .collect();
        words.sort_by_key(|w| std::cmp::Reverse(w.len()));
        Self { words }
    }

    pub fn apply(&self, text: &str) -> String {
        let mut out = text.to_string();
        let tokens: Vec<&str> = text.split_whitespace().collect();
        let token_set: HashSet<&str> = tokens.iter().copied().collect();

        for word in &self.words {
            // Exact phrase match first
            if out.contains(word) {
                continue;
            }
            // Fuzzy: if ASR output contains a token substring of hotword, replace
            for token in &token_set {
                if word.contains(token.as_bytes()) && token.len() >= 2 {
                    out = out.replace(token, word);
                }
            }
        }
        out
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn replaces_fuzzy_token_with_hotword() {
        let replacer = HotwordReplacer::from_lines(vec!["阿里巴巴".into()]);
        let result = replacer.apply("我去阿里上班");
        assert!(result.contains("阿里巴巴") || result.contains("阿里"));
    }
}
```

```bash
#!/usr/bin/env bash
# scripts/download-models.sh
set -euo pipefail
ROOT="$(cd "$(dirname "$0")/.." && pwd)"
MANIFEST="$ROOT/models/manifest.json"
mkdir -p "$ROOT/models/dev"

download_and_verify() {
  local url="$1" dest="$2"
  mkdir -p "$dest"
  local archive="$dest/archive"
  curl -L "$url" -o "$archive"
  echo "SHA256: $(shasum -a 256 "$archive" | awk '{print $1}')"
  case "$archive" in
    *.tar.bz2) tar -xjf "$archive" -C "$dest" ;;
    *.onnx) mv "$archive" "$dest/model.onnx" ;;
  esac
}

# Paraformer
download_and_verify \
  "https://github.com/k2-fsa/sherpa-onnx/releases/download/asr-models/sherpa-onnx-paraformer-zh-2024-03-09.tar.bz2" \
```

```rust
impl ModelManager {
    pub fn ensure_installed(&self, bundled: &Path) -> std::io::Result<ModelPaths> {
        let dest = self.models_dir();
        if !dest.exists() {
            std::fs::create_dir_all(&dest)?;
            copy_dir_all(bundled, &dest)?;
        }
        Ok(self.resolve_paths())
    }
}

fn copy_dir_all(src: &Path, dst: &Path) -> std::io::Result<()> {
    std::fs::create_dir_all(dst)?;
    for entry in std::fs::read_dir(src)? {
        let entry = entry?;
        let ty = entry.file_type()?;
        let to = dst.join(entry.file_name());
        if ty.is_dir() {
            copy_dir_all(&entry.path(), &to)?;
        } else {
            std::fs::copy(entry.path(), to)?;
        }
    }
    Ok(())
}
```

```rust
// src-tauri/src/asr/engine.rs
use sherpa_onnx::{OfflineRecognizer, OfflineRecognizerConfig, OfflineStream};
use std::path::Path;

pub struct AsrEngine {
    recognizer: OfflineRecognizer,
}

impl AsrEngine {
    pub fn new(paraformer_dir: &Path, num_threads: i32) -> Self {
        let model = paraformer_dir.join("model.int8.onnx");
        let tokens = paraformer_dir.join("tokens.txt");
        let config = OfflineRecognizerConfig::paraformer(&model, &tokens, num_threads);
        let recognizer = OfflineRecognizer::new(&config).expect("create recognizer");
        Self { recognizer }
    }

    pub fn transcribe(&mut self, samples: &[f32], sample_rate: u32) -> String {
        let mut stream = OfflineStream::new();
        stream.accept_waveform(sample_rate, samples);
        self.recognizer.decode(&mut stream);
        stream.result().text
    }
}
```

Note: Adjust `OfflineRecognizerConfig::paraformer` to match actual `sherpa-onnx` 1.13.x API after `cargo doc --open`. The exact constructor names come from `rust-api-examples/offline-paraformer-decode-files.rs` in the sherpa-onnx repo.

```rust
// src-tauri/src/audio/capture.rs
use cpal::traits::{DeviceTrait, HostTrait, StreamTrait};
use std::sync::{Arc, Mutex};

pub struct AudioCapture {
    samples: Arc<Mutex<Vec<f32>>>,
    _stream: cpal::Stream,
}

impl AudioCapture {
    pub fn start(sample_rate: u32) -> Result<Self, String> {
        let host = cpal::default_host();
        let device = host.default_input_device().ok_or("no input device")?;
        let config = device
            .default_input_config()
            .map_err(|e| e.to_string())?;
        let samples = Arc::new(Mutex::new(Vec::<f32>::new()));
        let buf = samples.clone();
        let stream = device
            .build_input_stream(
                &config.into(),
                move |data: &[f32], _| {
                    buf.lock().unwrap().extend_from_slice(data);
                },
                |e| eprintln!("audio error: {e}"),
                None,
            )
            .map_err(|e| e.to_string())?;
        stream.play().map_err(|e| e.to_string())?;
        Ok(Self { samples, _stream: stream })
    }

    pub fn stop(self) -> Vec<f32> {
        drop(self._stream);
        let mut data = self.samples.lock().unwrap();
        std::mem::take(&mut *data)
    }
}
```

```rust
// src-tauri/src/pipeline/session.rs
use crate::asr::engine::AsrEngine;
use crate::asr::punctuation::PunctuationEngine;
use crate::audio::capture::AudioCapture;
use crate::config::AppConfig;
use crate::post::hotword::HotwordReplacer;
use crate::post::pipeline::post_process;

pub enum SessionState {
    Idle,
    Recording(AudioCapture),
}

pub struct VoiceSession {
    asr: AsrEngine,
    punc: PunctuationEngine,
    hotwords: HotwordReplacer,
    config: AppConfig,
    state: SessionState,
}

impl VoiceSession {
    pub fn on_hotkey_press(&mut self) -> Result<(), String> {
        if matches!(self.state, SessionState::Idle) {
            let capture = AudioCapture::start(self.config.audio.sample_rate)?;
            self.state = SessionState::Recording(capture);
        }
        Ok(())
    }

    pub fn on_hotkey_release(&mut self) -> Result<Option<String>, String> {
        let capture = match std::mem::replace(&mut self.state, SessionState::Idle) {
            SessionState::Recording(c) => c,
            SessionState::Idle => return Ok(None),
        };
        let samples = capture.stop();
        if samples.len() < (self.config.audio.min_speech_ms as usize * 16) {
            return Ok(None);
        }
        let raw = self.asr.transcribe(&samples, self.config.audio.sample_rate);
        if raw.trim().is_empty() {
            return Ok(None);
        }
        let punctuated = self.punc.punctuate(&raw);
        let final_text = post_process(
            &punctuated,
            &self.hotwords,
            self.config.hotword.enabled,
        );
        Ok(Some(final_text))
    }
}
```

```rust
// src-tauri/src/hotkey/listener.rs
use rdev::{listen, Event, EventType, Key};
use std::sync::mpsc::Sender;
use std::thread;
use std::time::Instant;

pub enum HotkeyEvent {
    Pressed,
    Released,
}

pub fn spawn_hotkey_listener(trigger: Key, tx: Sender<HotkeyEvent>) {
    thread::spawn(move || {
        let mut pressed_at: Option<Instant> = None;
        let callback = move |event: Event| {
            match event.event_type {
                EventType::KeyPress(key) if key == trigger => {
                    pressed_at = Some(Instant::now());
                    let _ = tx.send(HotkeyEvent::Pressed);
                }
                EventType::KeyRelease(key) if key == trigger => {
                    if let Some(t0) = pressed_at.take() {
                        if t0.elapsed().as_millis() >= 300 {
                            let _ = tx.send(HotkeyEvent::Released);
                        }
                    }
                }
                _ => {}
            }
        };
        if let Err(e) = listen(callback) {
            eprintln!("hotkey listener error: {e}");
        }
    });
}

pub fn key_from_name(name: &str) -> Key {
    match name {
        "RightAlt" => Key::AltGr,
        "LeftAlt" => Key::Alt,
        "RightCtrl" => Key::ControlRight,
        _ => Key::AltGr,
    }
}
```

```rust
// src-tauri/src/log/mod.rs
use std::fs::OpenOptions;
use std::io::Write;

pub struct Logger {
    path: PathBuf,
}

impl Logger {
    pub fn new(root: PathBuf) -> std::io::Result<Self> {
        std::fs::create_dir_all(&root)?;
        Ok(Self { path: root.join("vosi.log") })
    }

    pub fn info(&self, msg: &str) {
        self.write("INFO", msg);
    }

    pub fn error(&self, msg: &str) {
        self.write("ERROR", msg);
    }

    fn write(&self, level: &str, msg: &str) {
        // Never log recognized text content — only operational metadata
        if let Ok(mut f) = OpenOptions::new().create(true).append(true).open(&self.path) {
            let _ = writeln!(f, "[{level}] {msg}");
        }
    }
}
```
